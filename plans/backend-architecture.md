**Stack:** Node.js · Express · PostgreSQL · Prisma ORM · JWT Auth · Multer · Mock AI  
**Structure:** Monorepo — `/server` directory alongside existing `/src` frontend  
**Port:** Backend `localhost:5000` · Frontend `localhost:3000` (CRA proxy)

## 3. API Endpoints

### 3.1 Authentication — `/api/v1/auth`

| Method | Path | Body | Response | Auth |
|--------|------|------|----------|------|
| `POST` | `/signup` | `{ name, email, password }` | `{ user, token }` | No |
| `POST` | `/login` | `{ email, password }` | `{ user, token }` | No |
| `POST` | `/logout` | — | `{ message }` | Yes |
| `GET` | `/me` | — | `{ user }` | Yes |
| `PATCH` | `/me` | `{ name?, avatarUrl?, calorieGoal?, proteinGoal?, carbsGoal?, fatGoal? }` | `{ user }` | Yes |

**Auth Flow:**
1. `POST /signup` → hash password with `bcrypt(12 rounds)` → create user → sign JWT (24h expiry) → return `{ user, token }`
2. `POST /login` → find user by email → compare password → sign JWT → return `{ user, token }`
3. All protected routes: `Authorization: Bearer <token>` header → [`auth.js`](server/src/middleware/auth.js) middleware decodes JWT, attaches `req.userId`
4. Frontend stores token in `localStorage` key `chefone_token`, user object in `chefone_user`

**JWT Payload:**
```json
{
  "userId": "uuid",
  "email": "user@example.com",
  "iat": 1715000000,
  "exp": 1715086400
}
```

## 5. Authentication Implementation

### Password Security
- **Hashing:** `bcrypt` with 12 salt rounds
- **JWT:** `jsonwebtoken` with HS256 algorithm, 24-hour expiry
- **Secret:** `JWT_SECRET` from `.env` (min 32 characters)

### Auth Middleware — [`server/src/middleware/auth.js`](server/src/middleware/auth.js)

```js
const authenticate = async (req, res, next) => {
  const header = req.headers.authorization;
  if (!header?.startsWith('Bearer ')) {
    return res.status(401).json({ success: false, error: 'No token provided' });
  }
  try {
    const token = header.split(' ')[1];
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    const user = await prisma.user.findUnique({ where: { id: decoded.userId } });
    if (!user) return res.status(401).json({ success: false, error: 'User not found' });
    req.userId = user.id;
    req.user = user;
    next();
  } catch (err) {
    return res.status(401).json({ success: false, error: 'Invalid token' });
  }
};
```

## 6. API Response Format

All endpoints return a consistent JSON structure:

```json
// Success
{
  "success": true,
  "data": { ... },
  "meta": {                    // Only for paginated responses
    "page": 1,
    "limit": 20,
    "total": 45,
    "totalPages": 3
  }
}

// Error
{
  "success": false,
  "error": "Descriptive error message",
  "details": [                 // Only for validation errors
    { "field": "email", "message": "Email is required" }
  ]
}
```

**HTTP Status Codes:**
| Code | Usage |
|------|-------|
| `200` | Successful GET, PATCH, DELETE |
| `201` | Successful POST (resource created) |
| `400` | Validation error, bad request |
| `401` | Not authenticated (missing/invalid token) |
| `403` | Forbidden (accessing another user's data) |
| `404` | Resource not found |
| `409` | Conflict (duplicate email on signup) |
| `429` | Rate limit exceeded |
| `500` | Internal server error |

## 8. File Upload Handling

### Multer Configuration — [`server/src/middleware/upload.js`](server/src/middleware/upload.js)

```js
const storage = multer.diskStorage({
  destination: (req, file, cb) => {
    const folder = req.uploadType || 'general';  // 'pantry', 'calories', 'scanner'
    const dir = path.join(__dirname, '../../uploads', folder);
    fs.mkdirSync(dir, { recursive: true });
    cb(null, dir);
  },
  filename: (req, file, cb) => {
    const uniqueName = `${Date.now()}-${crypto.randomBytes(6).toString('hex')}${path.extname(file.originalname)}`;
    cb(null, uniqueName);
  },
});

const upload = multer({
  storage,
  limits: { fileSize: 10 * 1024 * 1024 },  // 10MB max
  fileFilter: (req, file, cb) => {
    const allowed = ['image/jpeg', 'image/png', 'image/webp'];
    cb(null, allowed.includes(file.mimetype));
  },
});
```

Uploaded images are served statically via `express.static('uploads')` at `/uploads/*`.

import axios from 'axios';

const api = axios.create({
  baseURL: '/api/v1',
  headers: { 'Content-Type': 'application/json' },
});

// Attach JWT token to every request
api.interceptors.request.use((config) => {
  const token = localStorage.getItem('chefone_token');
  if (token) config.headers.Authorization = `Bearer ${token}`;
  return config;
});

// Handle 401 → redirect to login
api.interceptors.response.use(
  (res) => res.data,
  (err) => {
    if (err.response?.status === 401) {
      localStorage.removeItem('chefone_token');
      localStorage.removeItem('chefone_user');
      window.location.href = '/auth/sign-in';
    }
    return Promise.reject(err.response?.data || err);
  }
);

export default api;

## 14. Error Handling Strategy

### Global Error Handler — [`server/src/middleware/errorHandler.js`](server/src/middleware/errorHandler.js)

```js
const errorHandler = (err, req, res, next) => {
  console.error(`[${new Date().toISOString()}] ${err.stack}`);

  // Prisma known errors
  if (err.code === 'P2002') {
    return res.status(409).json({
      success: false,
      error: 'A record with this value already exists',
      details: [{ field: err.meta?.target?.[0], message: 'Must be unique' }]
    });
  }
  if (err.code === 'P2025') {
    return res.status(404).json({ success: false, error: 'Record not found' });
  }

  // Joi validation errors
  if (err.isJoi) {
    return res.status(400).json({
      success: false,
      error: 'Validation failed',
      details: err.details.map(d => ({ field: d.path[0], message: d.message }))
    });
  }

  // JWT errors
  if (err.name === 'JsonWebTokenError') {
    return res.status(401).json({ success: false, error: 'Invalid token' });
  }
  if (err.name === 'TokenExpiredError') {
    return res.status(401).json({ success: false, error: 'Token expired' });
  }

  // Multer errors
  if (err.code === 'LIMIT_FILE_SIZE') {
    return res.status(400).json({ success: false, error: 'File too large (max 10MB)' });
  }

  // Default
  res.status(err.statusCode || 500).json({
    success: false,
    error: process.env.NODE_ENV === 'production' ? 'Internal server error' : err.message
  });
};
```

### Async Handler Wrapper

```js
// utils/asyncHandler.js
const asyncHandler = (fn) => (req, res, next) =>
  Promise.resolve(fn(req, res, next)).catch(next);
```

Every controller function is wrapped with `asyncHandler` so thrown errors (including Prisma errors) automatically reach the global error handler.
